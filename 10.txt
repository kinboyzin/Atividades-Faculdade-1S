package fundamentos;

import javax.swing.JOptionPane;

public class FACULDADE_ATV10 {
public static void main(String[] args) {
	while(true) {
		String strNum =  JOptionPane.showInputDialog("Digite um n√∫mero inteiro, ou digite -999 para parar: ");
		
		int num = Integer.parseInt(strNum);
		
		if(num == -999) {
			break;
		}
	
		for(int i = 1; i <= num; i++) {
			if (num % i == 0) {
				System.out.println(i + " ");
				
			}
		}
		System.out.println(" ");
	}
}
}
